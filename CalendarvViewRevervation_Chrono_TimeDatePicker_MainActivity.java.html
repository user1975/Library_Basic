3    import android.graphics.Color; 
4    import android.os.Bundle; 
5    import android.os.SystemClock; 
6    import android.support.v7.app.AppCompatActivity; 
7    import android.view.View; 
8    import android.widget.Button; 
9    import android.widget.CalendarView; 
10   import android.widget.Chronometer; 
11   import android.widget.RadioButton; 
12   import android.widget.TextView; 
13   import android.widget.TimePicker; 
14    
15   public class MainActivity extends AppCompatActivity { 
16       Chronometer chrono; 
17       Button btnStart, btnEnd; 
18       RadioButton rdoCal, rdoTime; 
19       CalendarView calView; 
20       TimePicker tPicker; 
21       TextView tvYear, tvMonth, tvDay, tvHour, tvMinute; 
22       int selectYear, selectMonth, selectDay; 
23    
24       @Override 
25       public void onCreate(Bundle savedInstanceState) { 
26           super.onCreate(savedInstanceState); 
27           setContentView(R.layout.activity_main); 
28    
29           setTitle("시간 예약"); 
30    
31           // 버튼 
32           btnStart = (Button) findViewById(R.id.btnStart); 
33           btnEnd = (Button) findViewById(R.id.btnEnd); 
34    
35           // 크로노미터 
36           chrono = (Chronometer) findViewById(R.id.chronometer1); 
37    
38           // 라디오버튼 2개 
39           rdoCal = (RadioButton) findViewById(R.id.rdoCal); 
40           rdoTime = (RadioButton) findViewById(R.id.rdoTime); 
41    
42           // FrameLayout의 2개 위젯 
43           tPicker = (TimePicker) findViewById(R.id.timePicker1); 
44           calView = (CalendarView) findViewById(R.id.calendarView1); 
45    
46           // 텍스트뷰 중에서 연,월,일,시,분 숫자 
47           tvYear = (TextView) findViewById(R.id.tvYear); 
48           tvMonth = (TextView) findViewById(R.id.tvMonth); 
49           tvDay = (TextView) findViewById(R.id.tvDay); 
50           tvHour = (TextView) findViewById(R.id.tvHour); 
51           tvMinute = (TextView) findViewById(R.id.tvMinute); 
52    
53           // 초기화(invisible) 
54           tPicker.setVisibility(View.INVISIBLE); 
55           calView.setVisibility(View.INVISIBLE); 
56    
57           //선택시 처리 
58           rdoCal.setOnClickListener(new View.OnClickListener() { 
59               public void onClick(View v) { 
60                   tPicker.setVisibility(View.INVISIBLE); 
61                   calView.setVisibility(View.VISIBLE); 
62               } 
63           }); 
64    
65           rdoTime.setOnClickListener(new View.OnClickListener() { 
66               public void onClick(View v) { 
67                   tPicker.setVisibility(View.VISIBLE); 
68                   calView.setVisibility(View.INVISIBLE); 
69               } 
70           }); 
71    
72           // 타이머 설정 
73           btnStart.setOnClickListener(new View.OnClickListener() { 
74               public void onClick(View v) { 
75                   chrono.setBase(SystemClock.elapsedRealtime()); 
76                   chrono.start(); 
77                   chrono.setTextColor(Color.RED); 
78               } 
79           }); 
80    
81           // 버튼을 클릭하면 날짜,시간을 가져온다. 
82           btnEnd.setOnClickListener(new View.OnClickListener() { 
83               public void onClick(View v) { 
84                   chrono.stop(); 
85                   chrono.setTextColor(Color.BLUE); 
86    
87                   tvYear.setText(Integer.toString(selectYear)); 
88                   tvMonth.setText(Integer.toString(selectMonth)); 
89                   tvDay.setText(Integer.toString(selectDay)); 
90    
91                   tvHour.setText(Integer.toString(tPicker.getCurrentHour())); 
92                   tvMinute.setText(Integer.toString(tPicker.getCurrentMinute())); 
93    
94    
95               } 
96           }); 
97    
98           calView.setOnDateChangeListener(new CalendarView.OnDateChangeListener() { 
99               @Override 
100              public void onSelectedDayChange(CalendarView view, int year, int month, int dayOfMonth) { 
101   
102                  selectYear=year; 
103                  selectMonth=month +1; 
104                  selectDay = dayOfMonth; 
105   
106              } 
107          }); 
108   
109      } 
110  }
